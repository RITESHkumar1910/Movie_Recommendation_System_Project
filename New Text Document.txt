def convert(obj):

def convert3(obj):
    L = []
    counter = 0
    for i in ast.literal_eval(obj):
        if counter != 3:
            L.append(i["name"])

        else:
            break    
        return L

import ast

def clean_cast(x):
    if isinstance(x, str):
        try:
            x = ast.literal_eval(x)
        except:
            x = x.split(",")
    if isinstance(x, list):
        return [i.replace(" ", "") for i in x]
    return []
